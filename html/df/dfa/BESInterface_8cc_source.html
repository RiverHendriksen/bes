97 // This function uses the static variables timeout_jump_valid and timeout_jump
98 // The code looks at the value of BES.TimeOutInSeconds and/or the timeout
99 // context sent in the current request and, if that is greater than zero,
100 // uses that as the maximum amount of time for the request. The system alarm
101 // is set and this function is registered as the handler. If timeout_jump_valid
102 // is true, then it will use longjmp() (yes, really...) to end the request. Look
103 // below in execute_request() for the call to setjump() to see how this works.
104 // See the SIGWAIT code that's commented out below for an alternative impl.
105 // jhrg 5/31/16
106 static void catch_sig_alarm(int sig)
107 {
108  if (sig == SIGALRM) {
109  LOG("Child listener timeout after " << bes_timeout << " seconds, exiting." << endl);
110 
111  // Causes setjmp() below to return 1; see the call to
112  // execute_data_request_plan() in execute_request() below.
113  // jhrg 12/29/15
114  if (timeout_jump_valid)
115  longjmp(timeout_jump, 1);
116  else {
117  // This is the old version of this code; it forces the BES child
118  // listener to exit without returning an error message to the
119  // OLFS/client. jhrg 12/29/15
120  signal(SIGTERM, SIG_DFL);
121  raise(SIGTERM);
122  }
123  }
124 }

126 static void register_signal_handler()
127 {
128  struct sigaction act;
129  sigemptyset(&act.sa_mask);
130  sigaddset(&act.sa_mask, SIGALRM);
131  act.sa_flags = 0;
132 
133  // Note that we do not set SA_RESTART so an interrupted system call
134  // will return with an error and errno set to EINTR.
135 
136  act.sa_handler = catch_sig_alarm;
137  if (sigaction(SIGALRM, &act, 0))
138  throw BESInternalFatalError("Could not register a handler to catch alarm/timeout.", __FILE__, __LINE__);
139 }

163 // An alternative to a function that catches the signal; use sigwait()
164 // in a child thread after marking the signal as blocked. When/if sigwait()
165 // returns, look at the signal number and if it is the alarm, sort out
166 // what to do (throw an exception, ...). NB: A signal handler cannot
167 // portably throw an exception, but this code can.

171 static void* alarm_wait(void * /* arg */)
172 {
173  BESDEBUG("bes", "Starting: " << __PRETTY_FUNCTION__ << endl);
174 
175  // block SIGALRM
176  sigset_t sigset;
177  sigemptyset(&sigset);
178  sigaddset(&sigset, SIGALRM);
179  sigprocmask(SIG_BLOCK, &sigset, NULL);
180 
181  // Might replace this with a while loop. Not sure about interactions
182  // with other signal processing code in the BES. jhrg 12/28/15
183  int sig;
184  int result = sigwait(&sigset, &sig);
185  if (result != 0) {
186  BESDEBUG("bes", "Fatal error establishing timeout: " << strerror(result) << endl);
187  throw BESInternalFatalError(string("Fatal error establishing timeout: ") + strerror(result), __FILE__, __LINE__);
188  }
189  else if (result == 0 && sig == SIGALRM) {
190  BESDEBUG("bes", "Timeout found in " << __PRETTY_FUNCTION__ << endl);
191  throw BESTimeoutError("Timeout", __FILE__, __LINE__);
192  }
193  else {
194  stringstream oss;
195  oss << "While waiting for a timeout, found signal '" << result << "' in " << __PRETTY_FUNCTION__ << ends;
196  BESDEBUG("bes", oss.str() << endl);
197  throw BESInternalFatalError(oss.str(), __FILE__, __LINE__);
198  }
199 }

290 int BESInterface::execute_request(const string &from)
291 {
292  BESDEBUG("bes", "Entering: " << __PRETTY_FUNCTION__ << endl);
293 
294  if (!_dhi) {
295  throw BESInternalError("DataHandlerInterface can not be null", __FILE__, __LINE__);
296  }
297 
298  BESStopWatch sw;
299  if (BESISDEBUG(TIMING_LOG)) {
300  sw.start("BESInterface::execute_request", _dhi->data[REQUEST_ID]);
301 
302  bes_timing::elapsedTimeToReadStart = new BESStopWatch();
303  bes_timing::elapsedTimeToReadStart->start("TIME_TO_READ_START", _dhi->data[REQUEST_ID]);
304 
305  bes_timing::elapsedTimeToTransmitStart = new BESStopWatch();
306  bes_timing::elapsedTimeToTransmitStart->start("TIME_TO_TRANSMIT_START", _dhi->data[REQUEST_ID]);
307  }
308 
309  if (!_dhi) {
310  throw BESInternalError("DataHandlerInterface can not be null", __FILE__, __LINE__);
311  }
312 
313  _dhi->set_output_stream(_strm);
314  _dhi->data[REQUEST_FROM] = from;
315 
316  pid_t thepid = getpid();
317  ostringstream ss;
318  ss << thepid;
319  _dhi->data[SERVER_PID] = ss.str();
320 
321  // This is never used except as an arg to finish. jhrg 12/23/15
322  //int status = 0;
323 
324  // We split up the calls for the reason that if we catch an
325  // exception during the initialization, building, execution, or response
326  // transmit of the request then we can transmit the exception/error
327  // information.
328  try {
329  initialize();
330 
331  string m = BESLog::mark;
332  *(BESLog::TheLog()) << _dhi->data[REQUEST_FROM] << m <<
333  "request received" << m << endl;
334  BESLog::TheLog()->flush_me();
335 
336  // This does not do anything here or in BESBasicInterface or BESXMLInterface.
337  // Remove it? jhrg 12/23/15
338  validate_data_request();
339 
340  build_data_request_plan();
341 
342  if (!_transmitter)
343  throw BESInternalError("Unable to transmit the response, no transmitter", __FILE__, __LINE__);
344 
345  // This method does two key things: Calls the request handler to make a
346  // 'response object' (the C++ object that will hold the response) and
347  // then calls the transmitter to actually send it or build and send it.
348  //
349  // The timeout is also set in execute_data_request_plan(). The alarm signal
350  // handler (above), run when the timeout expires, will call longjmp with a
351  // return value of 1.
352  if (setjmp(timeout_jump) == 0) {
353  timeout_jump_valid = true;
354  execute_data_request_plan();
355  // Once we exit the block where setjmp() was called, the jump_buf is not valid
356  timeout_jump_valid = false;
357  }
358  else {
359  ostringstream oss;
360  oss << "BES listener timeout after " << bes_timeout << " seconds." << ends;
361  throw BESTimeoutError(oss.str(), __FILE__, __LINE__);
362  }
363 
364  _dhi->executed = true;
365  }
366  catch (BESError & ex) {
367  timeout_jump_valid = false;
368  return exception_manager(ex);
369  }
370  catch (bad_alloc &e) {
371  timeout_jump_valid = false;
372  BESInternalFatalError ex(string("BES out of memory: ") + e.what(), __FILE__, __LINE__);
373  return exception_manager(ex);
374  }
375  catch (exception &e) {
376  timeout_jump_valid = false;
377  BESInternalFatalError ex(string("C++ Exception: ") + e.what(), __FILE__, __LINE__);
378  return exception_manager(ex);
379  }
380  catch (...) {
381  timeout_jump_valid = false;
382  BESInternalError ex("An undefined exception has been thrown", __FILE__, __LINE__);
383  return exception_manager(ex);
384  }
385 
386  delete bes_timing::elapsedTimeToReadStart;
387  bes_timing::elapsedTimeToReadStart = 0;
388 
389  delete bes_timing::elapsedTimeToTransmitStart;
390  bes_timing::elapsedTimeToTransmitStart = 0;
391 
392  return finish(0 /* status */);;
393 }

497 void BESInterface::initialize()
498 {
499  // dhi has not been filled in at this point, so let's set a default
500  // transmitter given the protocol. The transmitter might change after
501  // parsing a request and given a return manager to use. This is done in
502  // build_data_plan.
503  //
504  // The reason I moved this from the build_data_plan method is because a
505  // registered initialization routine might throw an exception and we
506  // will need to transmit the exception info, which needs a transmitter.
507  // If an exception happens before this then the exception info is just
508  // printed to cout (see BESInterface::transmit_data()). -- pcw 09/05/06
509  BESDEBUG("bes", "Finding " << BASIC_TRANSMITTER << " transmitter ... " << endl);
510 
511  _transmitter = BESReturnManager::TheManager()->find_transmitter( BASIC_TRANSMITTER);
512  if (!_transmitter) {
513  string s = (string) "Unable to find transmitter " + BASIC_TRANSMITTER;
514  throw BESInternalError(s, __FILE__, __LINE__);
515  }
516  BESDEBUG("bes", "OK" << endl);
517 
518  BESStopWatch sw;
519  if (BESISDEBUG(TIMING_LOG)) sw.start("BESInterface::initialize", _dhi->data[REQUEST_ID]);
520 
521  BESDEBUG("bes", "Initializing request: " << _dhi->data[DATA_REQUEST] << " ... " << endl);
522  bool do_continue = true;
523  init_iter i = _init_list.begin();
524 
525  for (; i != _init_list.end() && do_continue == true; i++) {
526  p_bes_init p = *i;
527  do_continue = p(*_dhi);
528  }
529 
530  if (!do_continue) {
531  BESDEBUG("bes", "FAILED" << endl);
532  string se = "Initialization callback failed, exiting";
533  throw BESInternalError(se, __FILE__, __LINE__);
534  }
535  else {
536  BESDEBUG("bes", "OK" << endl);
537  }
538 }

544  // The derived class build_data_request_plan should be run first to

588 void BESInterface::execute_data_request_plan()
589 {
590  if (BESLog::TheLog()->is_verbose()) {
591  *(BESLog::TheLog()) << _dhi->data[SERVER_PID] << " from " << _dhi->data[REQUEST_FROM] << " ["
592  << _dhi->data[DATA_REQUEST] << "] executing" << endl;
593  }
594 
595  BESStopWatch sw;
596  if (BESISDEBUG(TIMING_LOG))
597  sw.start("BESInterface::execute_data_request_plan(\"" + _dhi->data[DATA_REQUEST] + "\")",
598  _dhi->data[REQUEST_ID]);
599 
600  // Set timeout if the 'bes_timeout' context value was passed in with the
601  // command.
602  bool found = false;
603  string context = BESContextManager::TheManager()->get_context("bes_timeout", found);
604  if (found) {
605  bes_timeout = strtol(context.c_str(), NULL, 10);
606  VERBOSE("Set request timeout to " << bes_timeout << " seconds (from context)." << endl);
607  alarm(bes_timeout);
608  }
609  else if (_timeout_from_keys != 0) {
610  bes_timeout = _timeout_from_keys;
611  VERBOSE("Set request timeout to " << bes_timeout << " seconds (from keys)." << endl);
612  alarm(bes_timeout);
613  }
614 
615 
616  BESDEBUG("bes", "Executing request: " << _dhi->data[DATA_REQUEST] << " ... " << endl);
617  BESResponseHandler *rh = _dhi->response_handler;
618  if (rh) {
619  rh->execute(*_dhi);
620  }
621  else {
622  BESDEBUG("bes", "FAILED" << endl);
623  string se = "The response handler \"" + _dhi->action + "\" does not exist";
624  throw BESInternalError(se, __FILE__, __LINE__);
625  }
626  BESDEBUG("bes", "OK" << endl);
627 
628  // Now we need to do the post processing piece of executing the request
629  invoke_aggregation();
630 
631  // And finally, transmit the response of this request
632  transmit_data();
633 
634  // Only clear the timeout if it has been set.
635  if (bes_timeout != 0) {
636  bes_timeout = 0;
637  alarm(0);
638  }
639 }

643 void BESInterface::invoke_aggregation()
644 {
645  if (_dhi->data[AGG_CMD] == "") {
646  if (BESLog::TheLog()->is_verbose()) {
647  *(BESLog::TheLog()) << _dhi->data[SERVER_PID] << " from " << _dhi->data[REQUEST_FROM] << " ["
648  << _dhi->data[DATA_REQUEST] << "]" << " not aggregating, command empty" << endl;
649  }
650  }
651  else {
652  BESAggregationServer *agg = BESAggFactory::TheFactory()->find_handler(_dhi->data[AGG_HANDLER]);
653  if (!agg) {
654  if (BESLog::TheLog()->is_verbose()) {
655  *(BESLog::TheLog()) << _dhi->data[SERVER_PID] << " from " << _dhi->data[REQUEST_FROM] << " ["
656  << _dhi->data[DATA_REQUEST] << "]" << " not aggregating, no handler" << endl;
657  }
658  }
659  else {
660  if (BESLog::TheLog()->is_verbose()) {
661  *(BESLog::TheLog()) << _dhi->data[SERVER_PID] << " from " << _dhi->data[REQUEST_FROM] << " ["
662  << _dhi->data[DATA_REQUEST] << "] aggregating" << endl;
663  }
664  }
665  }
666 
667  BESStopWatch sw;
668  if (BESISDEBUG(TIMING_LOG)) sw.start("BESInterface::invoke_aggregation", _dhi->data[REQUEST_ID]);
669 
670  if (_dhi->data[AGG_CMD] != "") {
671  BESDEBUG("bes", "aggregating with: " << _dhi->data[AGG_CMD] << " ... "<< endl);
672  BESAggregationServer *agg = BESAggFactory::TheFactory()->find_handler(_dhi->data[AGG_HANDLER]);
673  if (agg) {
674  agg->aggregate(*_dhi);
