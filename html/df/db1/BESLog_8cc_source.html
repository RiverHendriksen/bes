67 BESLog::BESLog() :
68  _flushed(1), _file_buffer(0), _suspended(0), _verbose(false)
69 {
70  _suspended = 0;
71  bool found = false;
72  try {
73  TheBESKeys::TheKeys()->get_value("BES.LogName", _file_name, found);
74  }
75  catch (...) {
76  string err = (string) "BES Fatal: unable to determine log file name."
77  + " The key BES.LogName has multiple values";
78  cerr << err << endl;
79  throw BESInternalFatalError(err, __FILE__, __LINE__);
80  }
81  if (_file_name == "") {
82  string err = (string) "BES Fatal: unable to determine log file name."
83  + " Please set BES.LogName in your initialization file";
84  cerr << err << endl;
85  throw BESInternalFatalError(err, __FILE__, __LINE__);
86  }
87  _file_buffer = new ofstream(_file_name.c_str(), ios::out | ios::app);
88  if (!(*_file_buffer)) {
89  string err = (string) "BES Fatal; cannot open log file " + _file_name + ".";
90  cerr << err << endl;
91  throw BESInternalFatalError(err, __FILE__, __LINE__);
92  }
93  found = false;
94  string verbose;
95  TheBESKeys::TheKeys()->get_value("BES.LogVerbose", verbose, found);
96  if (verbose == "YES" || verbose == "Yes" || verbose == "yes") {
97  _verbose = true;
98  }
99 }

118 void BESLog::dump_time()
119 {
120  const time_t sctime = time(NULL);
121  const struct tm *sttime = localtime(&sctime);
122  char zone_name[10];
123  strftime(zone_name, sizeof(zone_name), "%Z", sttime);
124  char *b = asctime(sttime);
125  (*_file_buffer) << mark << zone_name << " ";
126  for (register int j = 0; b[j] != '\n'; j++)
127  (*_file_buffer) << b[j];
128  pid_t thepid = getpid();
129  (*_file_buffer) << mark << "pid: " << thepid << mark;
130  _flushed = 0;
131 }
