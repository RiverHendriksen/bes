60 void BESXMLCommand::set_response()
61 {
62  _dhi.response_handler = BESResponseHandlerList::TheList()->find_handler(_dhi.action);
63  if (!_dhi.response_handler) {
64  throw BESSyntaxUserError(string("Command '") + _dhi.action + "' does not have a registered response handler",
65  __FILE__, __LINE__);
66  }
67 
68  // The _str_cmd is a text version of the xml command used for the log.
69  // It is not used for anything else. I think the 'sql like' syntax is
70  // actually no longer used by the BES and that the software in cmdln
71  // translates that syntax into XML. But I'm not 100% sure... jhrg 12/29/15
72  _dhi.data[DATA_REQUEST] = _str_cmd;
73 
74  BESLog::TheLog()->flush_me();
75  string m = BESLog::mark;
76  *(BESLog::TheLog()) << _dhi.data[REQUEST_FROM] << m
77  << _str_cmd << m <<
78  "request received" << m << endl;
79  BESLog::TheLog()->flush_me();
80 }
