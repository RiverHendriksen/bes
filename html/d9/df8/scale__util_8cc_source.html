107 bool monotonic_and_uniform(const vector<double> &values, double res)
108 {
109  vector<double>::size_type end_index = values.size() - 1;
110  for (vector<double>::size_type i = 0; i < end_index; ++i) {
111  BESDEBUG(DEBUG_KEY, "values[" << i+1 << "]: " << values[i+1] << " - values[" << i << "]: " << values[i] << endl);
112  BESDEBUG(DEBUG_KEY, values[i+1] - values[i] <<" != res: " << res << endl);
113  if (!same_as((values[i+1] - values[i]), res))
114  return false;
115  }
116 
117  return true;
118 }

131 vector<double> get_geotransform_data(Array *x, Array *y, bool test_maps /* default: false*/)
132 {
133 #ifndef NDEBUG
134  test_maps = true;
135 #endif
136 
137  SizeBox size = get_size_box(x, y);
138 
139  y->read();
140  vector<double> y_values(size.y_size);
141  extract_double_array(y, y_values);
142 
143  double res_y = (y_values[y_values.size()-1] - y_values[0]) / (y_values.size() -1);
144 
145  if (test_maps && !monotonic_and_uniform(y_values, res_y)){
146  string msg = "The grids maps/dimensions must be monotonic and uniform (" + y->name() + ").";
147  BESDEBUG(DEBUG_KEY,"ERROR get_geotransform_data(): " << msg << endl);
148  throw BESError(msg,BES_SYNTAX_USER_ERROR,__FILE__,__LINE__);
149  }
150  x->read();
151  vector<double> x_values(size.x_size);
152  extract_double_array(x, x_values);
153 
154  double res_x = (x_values[x_values.size()-1] - x_values[0]) / (x_values.size() -1);
155 
156  if (test_maps && !monotonic_and_uniform(x_values, res_x)){
157  string msg = "The grids maps/dimensions must be monotonic and uniform (" + x->name() + ").";
158  BESDEBUG(DEBUG_KEY,"ERROR get_geotransform_data(): " << msg << endl);
159  throw BESError(msg,BES_SYNTAX_USER_ERROR,__FILE__,__LINE__);
160 }
161  // Xgeo = GT(0) + Xpixel*GT(1) + Yline*GT(2)
162  // Ygeo = GT(3) + Xpixel*GT(4) + Yline*GT(5)
163 
164  // Original comment:
165  // In case of north up images, the GT(2) and GT(4) coefficients are zero,
166  // and the GT(1) is pixel width, and GT(5) is pixel height. The (GT(0),GT(3))
167  // position is the top left corner of the top left pixel of the raster.
168  //
169  // Note that for an inverted dataset, use min_y and res_y for GT(3) and GT(5)
170  //
171  // 10/27/16 I decided to not treat this as lat/lon information but simply
172  // develop a mathematical transform that will be correct for the data as given
173  // _so long as_ the data are monotonic and uniform. jhrg
174 
175  vector<double> geo_transform(6);
176  geo_transform[0] = x_values[0];
177  geo_transform[1] = res_x;
178  geo_transform[2] = 0; // Assumed because the x/y maps are vectors
179  geo_transform[3] = y_values[0];
180  geo_transform[4] = 0;
181  geo_transform[5] = res_y;
182 
183  return geo_transform;
184 }

197 vector<GDAL_GCP> get_gcp_data(Array *x, Array *y, int sample_x, int sample_y)
198 {
199  SizeBox size = get_size_box(x, y);
200 
201  y->read();
202  vector<double> y_values(size.y_size);
203  extract_double_array(y, y_values);
204 
205  x->read();
206  vector<double> x_values(size.x_size);
207  extract_double_array(x, x_values);
208 
209  // Build the GCP list.
210 
211  // Determine how many control points to use. Subset by a factor of M
212  // but never use less than 10% of of the x and y axis values (each)
213  // FIXME Just use given values for now, which will default to 1.
214 
215  // Build the GCP list, sampling as per sample_x and sample_y
216  unsigned long n_gcps = (size.x_size/sample_x) * (size.y_size/sample_y);
217 
218  vector<GDAL_GCP> gcp_list(n_gcps);
219  GDALInitGCPs(n_gcps, &gcp_list[0]); // allocates the 'list'; free with Deinit
220 
221  unsigned long count = 0;
222  for (int i = 0; i < size.x_size; i += sample_x) {
223  // The test for count < n_gcps corrects for discrepencies between integer
224  // division and the simple increment used by the loops. 3/29/17 jhrg
225  for (int j = 0; count < n_gcps && j < size.y_size; j += sample_y) {
226  gcp_list[count].dfGCPLine = j;
227  gcp_list[count].dfGCPPixel = i;
228  gcp_list[count].dfGCPX = x_values[i];
229  gcp_list[count].dfGCPY = y_values[j];
230 
231  ++count;
232  }
233  }
234 
235  return gcp_list;
236 }

316  unsigned long y = band->GetYSize();

322

331  case dods_int16_c:

333  break;
334  case dods_uint32_c:
335  return transfer_values_helper<dods_uint32>(source->GetRasterBand(1), x, y, result);
336  break;
337  case dods_int32_c:
338  return transfer_values_helper<dods_int32>(source->GetRasterBand(1), x, y, result);
339  break;
340  case dods_float32_c:
341  return transfer_values_helper<dods_float32>(source->GetRasterBand(1), x, y, result);
342  break;
343  case dods_float64_c:
344  return transfer_values_helper<dods_float64>(source->GetRasterBand(1), x, y, result);
345  break;
346  case dods_uint8_c:

349  case dods_int8_c:

353  case dods_int64_c:

525 bool array_is_effectively_2D(const libdap::Array *src)
526 {
527  Array *a = const_cast<Array*>(src);
528  int numDims = a->dimensions();
529  if (numDims == 2) return true;
530  if (numDims < 2) return false;
531 
532  Array::Dim_iter xDim = get_x_dim(a);
533  for (Array::Dim_iter thisDim = a->dim_begin(); thisDim < xDim; thisDim++) {
534  unsigned long size = a->dimension_size(thisDim, true);
535  if (size > 1) {
536  return false;
537  }
538  }
539  return true;

642 auto_ptr<GDALDataset> build_src_dataset(Array *data, Array *x, Array *y, const string &srs)
643 {
644  GDALDriver *driver = GetGDALDriverManager()->GetDriverByName("MEM");
645  if(!driver){
646  string msg = string("Could not get the Memory driver for GDAL: ") + CPLGetLastErrorMsg();
647  BESDEBUG(DEBUG_KEY, "ERROR build_src_dataset(): " << msg << endl);
648  throw BESError(msg,BES_INTERNAL_ERROR,__FILE__,__LINE__);
649  }
650  SizeBox array_size = get_size_box(x, y);
651 
652  // The MEM driver takes no creation options jhrg 10/6/16
653  auto_ptr<GDALDataset> ds(driver->Create("result", array_size.x_size, array_size.y_size,
654  1 /* nBands*/, get_array_type(data), NULL /* driver_options */));
655 
656 #if ADD_BAND
657  add_band_data(data, ds.get());
658 #endif
659 
660  // Get the one band for this dataset and load it with data
661  GDALRasterBand *band = ds->GetRasterBand(1);
662  if (!band){
663  string msg = "Could not get the GDAL RasterBand for Array '" + data->name() + "': " + CPLGetLastErrorMsg();
664  BESDEBUG(DEBUG_KEY,"ERROR build_src_dataset(): " << msg << endl);
665  throw BESError(msg,BES_INTERNAL_ERROR,__FILE__,__LINE__);
666  }
667  // Set the no data value here; I'm not sure what the affect of using NaN will be... jhrg 10/11/16
668  double no_data = get_missing_data_value(data);
669  band->SetNoDataValue(no_data);
670 
671 #if !ADD_BAND
672  read_band_data(data, band);
673 #endif
674 
675  vector<double> geo_transform = get_geotransform_data(x, y);
676  ds->SetGeoTransform(&geo_transform[0]);
677 
678  OGRSpatialReference native_srs;
679  if (CE_None != native_srs.SetWellKnownGeogCS(srs.c_str())){
680  string msg = "Could not set '" + srs + "' as the dataset native CRS.";
681  BESDEBUG(DEBUG_KEY,"ERROR build_src_dataset(): " << msg << endl);
682  throw BESError(msg,BES_SYNTAX_USER_ERROR,__FILE__,__LINE__);
683 }
684  // I'm not sure what to do about the Projected Coordinate system. jhrg 10/6/16
685  // native_srs.SetUTM( 11, TRUE );
686 
687  // Connect the SRS/CRS to the GDAL Dataset
688  char *pszSRS_WKT = NULL;
689  native_srs.exportToWkt( &pszSRS_WKT );
690  ds->SetProjection( pszSRS_WKT );
691  CPLFree( pszSRS_WKT );
692 
693  return ds;
694 }
